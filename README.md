## Install

```bash
npm install --save use-3d-effect
```

## Usage

```tsx
import * as React from 'react'

import { useMyHook } from 'use-3d-effect'

const Example = () => {
  const example = useMyHook()
  return (
    <div>
      {example}
    </div>
  )
}
```
